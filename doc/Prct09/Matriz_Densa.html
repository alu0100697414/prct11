def initialize(filas, columnas)
  super
  @pos = Array.new(@filas,0)
  i = 0
  while i < @filas
    @pos[i] = Array.new(@columnas,0)
    i += 1
  end
end

def num_nulos
  total = @filas*@columnas
  nn = 0

  i = 0
   while(i < @filas)
     j = 0
     while(j < @columnas)
       if(@pos[i][j] != 0)
         nn += 1
       end
       j += 1
     end
     i += 1
   end

  n = total - nn

  n.to_f/total.to_f
end

def to_s
  s = ""
  i = 0
  while(i < @filas)
    j = 0
    while(j < @columnas)
      s += "#{@pos[i][j].to_s} "
      j += 1
    end
    puts " "
    i += 1
  end
  s
end
